# Crie uma função recursiva que calcule a soma de todos os números numa lista.
numeros = [1, 2, 3, 4, 5]
def soma_recursiva(lista):
    if not lista:
        return 0
    else:
        return lista[0] + soma_recursiva(lista[1])

print(soma_recursiva(numeros))
